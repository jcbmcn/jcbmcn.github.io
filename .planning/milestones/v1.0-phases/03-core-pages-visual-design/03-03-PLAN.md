    import BaseLayout from '../../layouts/BaseLayout.astro';
    import { getCollection } from 'astro:content';

    export async function getStaticPaths() {
      const posts = await getCollection('blog', ({ data }) => data.draft !== true);
      return posts.map((entry) => ({
        params: { slug: entry.id },
        props: { entry },
      }));
    }

    const { entry } = Astro.props;
    const { Content } = await entry.render();

    const { title, date, tags = [], featuredImage, images } = entry.data;
    const image = featuredImage ?? images?.[0] ?? undefined;

    const formattedDate = date
      ? date.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })
      : null;

    <!-- Copy-to-clipboard: injected after render so all <pre> elements exist in DOM -->
    <script>
      function addCopyButtons() {
        const codeBlocks = document.querySelectorAll('pre');

        codeBlocks.forEach((pre) => {
          // Avoid double-adding if script runs twice
          if (pre.querySelector('.copy-btn')) return;

          const btn = document.createElement('button');
          btn.className = 'copy-btn';
          btn.setAttribute('aria-label', 'Copy code');
          btn.textContent = 'Copy';

          btn.addEventListener('click', async () => {
            const code = pre.querySelector('code');
            const text = code?.textContent ?? '';
            try {
              await navigator.clipboard.writeText(text);
              btn.textContent = 'Copied!';
              btn.classList.add('copy-btn--copied');
              setTimeout(() => {
                btn.textContent = 'Copy';
                btn.classList.remove('copy-btn--copied');
              }, 2000);
            } catch {
              btn.textContent = 'Failed';
              setTimeout(() => {
                btn.textContent = 'Copy';
              }, 2000);
            }
          });

          // Position button relative to pre
          pre.style.position = 'relative';
          pre.appendChild(btn);
        });
      }

      // Run after DOM is ready
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', addCopyButtons);
      } else {
        addCopyButtons();
      }
    </script>

    **Key implementation notes:**
    - The `<script>` block is a plain Astro script — Astro automatically bundles it. No `client:load` island needed since this page is fully server-rendered HTML.
    - The copy button uses `position: absolute` inside each `pre` block. The script sets `pre.style.position = 'relative'` to establish the containing block.
    - The prose class comes from `@tailwindcss/typography` loaded via `@plugin "@tailwindcss/typography"` in global.css.
    - Shiki renders code blocks as `<pre><code>` with inline `color` styles (not CSS classes). The copy button reads `code.textContent` to get plain text.
    - The copy button opacity is 0 by default and appears on `pre:hover` — standard UX pattern for code blocks.
    - `:global()` is required for styles targeting elements rendered inside `<Content />` since they are outside the scoped component.
    - `clamp(1.5rem, 4vw, 2rem)` for the title gives responsive sizing without a breakpoint.
