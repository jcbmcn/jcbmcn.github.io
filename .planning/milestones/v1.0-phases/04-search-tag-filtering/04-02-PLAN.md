    **Step 5 — Add `<script>` block with Fuse.js filter logic:**

    Add before `</BaseLayout>`:
    ```astro
    <script>
    import Fuse from 'fuse.js';

    document.addEventListener('DOMContentLoaded', () => {
      // --- Data ---
      const raw = document.getElementById('post-data')?.textContent ?? '[]';
      const postData: { id: string; title: string; tags: string[]; description: string }[] = JSON.parse(raw);

      const fuse = new Fuse(postData, {
        keys: ['title', 'tags', 'description'],
        ignoreLocation: true,
        threshold: 0.3,
        includeScore: false,
      });

      // --- DOM refs ---
      const searchInput = document.getElementById('search-input') as HTMLInputElement | null;
      const clearBtn = document.getElementById('clear-filters') as HTMLButtonElement | null;
      const emptyState = document.getElementById('empty-state') as HTMLParagraphElement | null;
      const filterStatus = document.getElementById('filter-status') as HTMLDivElement | null;
      const countEl = document.querySelector('.listing-count') as HTMLParagraphElement | null;
      const postCards = Array.from(document.querySelectorAll<HTMLElement>('[data-tags]'));
      const filterChips = Array.from(document.querySelectorAll<HTMLButtonElement>('.filter-chip'));

      // --- State ---
      let activeTags = new Set<string>();
      let searchQuery = '';

      // --- Core filter function ---
      function applyFilters() {
        // 1. Determine IDs matching full-text search
        let searchMatchIds: Set<string> | null = null;
        if (searchQuery.trim() !== '') {
          const results = fuse.search(searchQuery.trim());
          searchMatchIds = new Set(results.map(r => r.item.id));
        }

        // 2. Show/hide cards
        let visibleCount = 0;
        postCards.forEach(card => {
          const cardId = card.querySelector('a')?.getAttribute('href')?.replace(/^\/posts\//, '').replace(/\/$/, '') ?? '';
          const cardTagsRaw = card.getAttribute('data-tags') ?? '';
          const cardTags = cardTagsRaw ? cardTagsRaw.split(',') : [];

          // Check search match
          const matchesSearch = searchMatchIds === null || searchMatchIds.has(cardId);

          // Check tag match (AND logic — must match ALL active tags)
          const matchesTags = activeTags.size === 0 || [...activeTags].every(t => cardTags.includes(t));

          const visible = matchesSearch && matchesTags;
          card.style.display = visible ? '' : 'none';
          if (visible) visibleCount++;
        });

        // 3. Update post count
        if (countEl) {
          countEl.textContent = `${visibleCount} post${visibleCount !== 1 ? 's' : ''}`;
        }

        // 4. Show/hide empty state with correct copy per UI-SPEC copywriting contract
        if (emptyState) {
          if (visibleCount === 0) {
            const hasSearch = searchQuery.trim() !== '';
            const hasTags = activeTags.size > 0;
            if (hasSearch && hasTags) {
              emptyState.textContent = `No posts match "${searchQuery.trim()}" with the selected tags.`;
            } else if (hasSearch) {
              emptyState.textContent = `No posts match "${searchQuery.trim()}".`;
            } else if (activeTags.size === 1) {
              emptyState.textContent = `No posts tagged "${[...activeTags][0]}".`;
            } else {
              emptyState.textContent = `No posts match all selected tags.`;
            }
            emptyState.hidden = false;
          } else {
            emptyState.hidden = true;
          }
        }

        // 5. Update aria-live region
        if (filterStatus) {
          filterStatus.textContent = `${visibleCount} post${visibleCount !== 1 ? 's' : ''} shown`;
        }

        // 6. Show/hide clear button
        const hasActiveFilters = activeTags.size > 0 || searchQuery.trim() !== '';
        if (clearBtn) clearBtn.hidden = !hasActiveFilters;
      }

      // --- Sync PostCard tag chip active states ---
      function syncPostCardChips() {
        document.querySelectorAll<HTMLButtonElement>('.post-card__tag[data-tag]').forEach(btn => {
          const tag = btn.getAttribute('data-tag') ?? '';
          const isActive = activeTags.has(tag);
          btn.setAttribute('aria-pressed', String(isActive));
          btn.classList.toggle('post-card__tag--active', isActive);
        });
      }

      // --- Sync filter bar chip active states ---
      function syncFilterChips() {
        filterChips.forEach(chip => {
          const tag = chip.getAttribute('data-tag') ?? '';
          const isActive = activeTags.has(tag);
          chip.setAttribute('aria-pressed', String(isActive));
          chip.classList.toggle('filter-chip--active', isActive);
        });
      }

      // --- Toggle tag helper ---
      function toggleTag(tag: string) {
        if (activeTags.has(tag)) {
          activeTags.delete(tag);
        } else {
          activeTags.add(tag);
        }
        syncFilterChips();
        syncPostCardChips();
        applyFilters();
      }

      // --- Event: search input ---
      searchInput?.addEventListener('input', (e) => {
        searchQuery = (e.target as HTMLInputElement).value;
        applyFilters();
      });

      // --- Event: filter bar chip clicks ---
      filterChips.forEach(chip => {
        chip.addEventListener('click', () => {
          const tag = chip.getAttribute('data-tag') ?? '';
          if (tag) toggleTag(tag);
        });
      });

      // --- Event: PostCard tag chip clicks (delegated) ---
      document.addEventListener('click', (e) => {
        const btn = (e.target as HTMLElement).closest<HTMLButtonElement>('.post-card__tag[data-tag]');
        if (btn) {
          const tag = btn.getAttribute('data-tag') ?? '';
          if (tag) toggleTag(tag);
        }
      });

      // --- Event: Clear filters ---
      clearBtn?.addEventListener('click', () => {
        activeTags.clear();
        searchQuery = '';
        if (searchInput) searchInput.value = '';
        syncFilterChips();
        syncPostCardChips();
        applyFilters();
      });
    });
    </script>
    ```
