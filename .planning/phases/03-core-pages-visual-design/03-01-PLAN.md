    @plugin "@tailwindcss/typography";

    @layer base {
      :root {
        /* Color tokens — light mode */
        --color-bg: theme(colors.slate.50);
        --color-surface: theme(colors.white);
        --color-surface-alt: theme(colors.slate.100);
        --color-border: theme(colors.slate.200);
        --color-text-primary: theme(colors.slate.900);
        --color-text-secondary: theme(colors.slate.500);
        --color-text-muted: theme(colors.slate.400);
        --color-accent: #6366F1;
        --color-accent-hover: theme(colors.indigo.600);
        --color-accent-subtle: theme(colors.indigo.50);

        /* Typography tokens */
        --font-sans: 'Inter', system-ui, -apple-system, sans-serif;
      }

      /* Dark mode token overrides — values stubbed for v2 */
      /* [data-theme="dark"] { ... } */

      html {
        font-family: var(--font-sans);
        background-color: var(--color-bg);
        color: var(--color-text-primary);
      }

      body {
        margin: 0;
      }

      a {
        color: var(--color-accent);
      }

      a:hover {
        color: var(--color-accent-hover);
      }
    }

    Create `src/layouts/BaseLayout.astro`:

    ```astro
    ---
    interface Props {
      title?: string;
      description?: string;
    }

    const {
      title = 'Jacob McNeilly',
      description = 'Thoughts on SRE, Kubernetes, and building systems that actually work.',
    } = Astro.props;
    ---
    <!DOCTYPE html>
    <html lang="en">
      <head>
        <meta charset="utf-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1" />
        <title>{title}</title>
        <meta name="description" content={description} />
        <link rel="preconnect" href="https://fonts.googleapis.com" />
        <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
        <link
          href="https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&display=swap"
          rel="stylesheet"
        />
        <link rel="icon" type="image/svg+xml" href="/favicon.svg" />
        <link rel="icon" type="image/png" href="/favicon-96x96.png" sizes="96x96" />
        <link rel="apple-touch-icon" href="/apple-touch-icon.png" />
        <link rel="manifest" href="/site.webmanifest" />
      </head>
      <body>
        <slot />
      </body>
    </html>

    <style is:global>
      @import '../styles/global.css';
    </style>
    ```

    NOTE on Tailwind 4 import: In Tailwind 4 with the Vite plugin, the recommended pattern is to import global.css from within a layout or page using `<style is:global>@import '../styles/global.css';</style>` — OR to add it as a plain `<link>` import in astro.config.mjs. Either works; the style block approach is fine and self-contained per component. The `@import "tailwindcss"` at the top of global.css is the Tailwind 4 entry point (replaces `@tailwind base/components/utilities` from v3).
