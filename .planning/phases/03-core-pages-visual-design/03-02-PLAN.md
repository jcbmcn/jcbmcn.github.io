    ---
    import type { CollectionEntry } from 'astro:content';

    interface Props {
      entry: CollectionEntry<'blog'>;
    }

    const { entry } = Astro.props;
    const { title, date, tags = [], featuredImage, images } = entry.data;
    const image = featuredImage ?? images?.[0] ?? undefined;
    const slug = entry.id;

    const formattedDate = date
      ? date.toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })
      : null;
    ---

    <article class="post-card">
      <a href={`/posts/${slug}/`} class="post-card__link">
        <div class="post-card__body">
          <div class="post-card__text">
            <h2 class="post-card__title">{title}</h2>
            {formattedDate && (
              <time class="post-card__date" datetime={date?.toISOString()}>
                {formattedDate}
              </time>
            )}
            {tags.length > 0 && (
              <ul class="post-card__tags">
                {tags.map((tag) => (
                  <li class="post-card__tag">{tag}</li>
                ))}
              </ul>
            )}
          </div>
          {image && (
            <div class="post-card__image-wrap">
              <img
                src={image}
                alt=""
                width="96"
                height="96"
                class="post-card__image"
                loading="lazy"
                decoding="async"
              />
            </div>
          )}
        </div>
      </a>
    </article>
