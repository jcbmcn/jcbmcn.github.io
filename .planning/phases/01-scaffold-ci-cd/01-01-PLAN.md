**Step 1 — Create .github/workflows/deploy.yml** (per D-08: new file, do NOT edit hugo.yml):

```yaml
# .github/workflows/deploy.yml
name: Deploy to GitHub Pages

on:
  push:
    branches: [main]
  workflow_dispatch:

# Required for GitHub Pages deployment (per D-12)
permissions:
  contents: read
  pages: write
  id-token: write

# Keep existing concurrency settings — prevents partial deploys (per agent discretion)
concurrency:
  group: "pages"
  cancel-in-progress: false

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Install, build, and upload
        uses: withastro/action@v5
        with:
          node-version: 20  # Node.js 20 LTS pinned (per D-11)

  deploy:
    needs: build
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
```
